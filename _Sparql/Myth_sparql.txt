-- Olympian Gods and Goddess
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
prefix : <http://www.semanticweb.org/ontologies/mythologies#> 
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
Prefix g: <http://www.semanticweb.org/mythologies/>
SELECT ?god ?group ?label
WHERE {graph g:data {
 ?god a :Olympian; 
  OPTIONAL {?god a ?group . FILTER(?group = :OlympianGod || ?group = :OlympianGoddess)  } 
  OPTIONAL {?god skos:prefLabel ?label filter (langmatches(lang(?label), "en")) }
  }} order by ?group ?god

-- Olympian Gods and Goddess filter show only label that contains "ap" 
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
prefix : <http://www.semanticweb.org/ontologies/mythologies#> 
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
Prefix g: <http://www.semanticweb.org/mythologies/>
SELECT ?god ?group ?label
WHERE {graph g:data {
 ?god a :Olympian .
   ?god a ?group . FILTER(?group = :OlympianGod || ?group = :OlympianGoddess)  .
   ?god skos:prefLabel ?label filter (langmatches(lang(?label), "en") && regex(?label, "ap", "i"))  .
  }} order by ?group ?god



-- model list of classes
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
Prefix g: <http://www.semanticweb.org/mythologies/>
SELECT DISTINCT ?class ?subClassOf ?label ?description
WHERE {graph g:metamodel {
  ?class a owl:Class.
  OPTIONAL { ?class rdfs:subClassOf ?subClassOf}
  OPTIONAL { ?class rdfs:label ?label}
  OPTIONAL { ?class rdfs:comment ?description}
  }} order by ?class

-- model list of object properties
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
Prefix g: <http://www.semanticweb.org/mythologies/>
SELECT DISTINCT ?prop
WHERE {graph g:metamodel {
  ?prop a owl:ObjectProperty.
  }} order by ?prop

-- list of data properties
PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
Prefix g: <http://www.semanticweb.org/mythologies/>
SELECT DISTINCT ?prop ?subPropertyOf ?domain ?range
WHERE {graph g:metamodel {
  ?prop a owl:DatatypeProperty.
    OPTIONAL { ?prop rdfs:subPropertyOf ?subPropertyOf}
    OPTIONAL { ?prop rdfs:domain ?domain}
    OPTIONAL { ?prop rdfs:range ?range}
  }} order by ?prop

SELECT ?god ?group ?romanCounterPart ?husband
WHERE {graph <http://www.semanticweb.org/mythologies/data> {

-- list of entities part of scheme = Argonauts

select ?ent ?label ?wikidata
where { graph g:data { ?ent skos:inScheme sch:Argonauts; 
                       OPTIONAL {?ent skos:prefLabel ?label filter (langmatches(lang(?label), "fr")) }
                       OPTIONAL {?ent skos:exactMatch ?wikidata . FILTER(regex(str(?wikidata), "www.wikidata.org" ) )  }
  }
}
