PREFIX owl: <http://www.w3.org/2002/07/owl#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX skos: <http://www.w3.org/2004/02/skos/core#>
Prefix g: <http://www.semanticweb.org/mythologies/>
SELECT distinct ?p ?prop ?check
WHERE {graph g:data {
 ?s a skos:Concept; ?p ?o .
    OPTIONAL {graph g:metamodel {?p a ?prop} }
    BIND(BOUND(?prop) && strlen(?prop)>0 as ?check) FILTER (?check=false)
  }}

-- entity without gender over myth:HumanNonhumanBeing or myth:MythicAnimals 
prefix rdfs: <http://www.w3.org/2000/01/rdf-schema#>

select ?class ?ent ?gender ?check
where {graph g:metamodel {?class rdfs:subClassOf* myth:HumanNonhumanBeing . 
    {graph g:data {?ent a ?class . OPTIONAL{?ent p:hasGender ?gender}
     BIND(BOUND(?gender) as ?check) FILTER (?check=false) }
    }
   } 
}  
order by ?ent ?class
